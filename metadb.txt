DROP TABLE IF EXISTS categorical_metadata;
DROP TABLE IF EXISTS numerical_metadata;

CREATE TABLE categorical_metadata (
    attribute VARCHAR(255),
    value VARCHAR(255),
    idf REAL,
    qf REAL,
    similarity VARCHAR(255)
    PRIMARY KEY (attribute, value)
);

CREATE TABLE numerical_metadata (
    attribute VARCHAR(255),
    value real,
    idf REAL,
    qf REAL,
    PRIMARY KEY (attribute, value)
);
